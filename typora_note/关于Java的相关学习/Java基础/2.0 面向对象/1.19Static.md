package StaticTest;




/*
*  1，static 静态的
*
*   2。static 可以用来修饰属性方法 代码块 内部类
*
*   3.使用static修饰属性：静态变量 或者叫做类变量
*
*   3。1属性 安是否使用static修饰，又分为：静态属性 vs 非静态属性（实例变量）
*           实例变量：我们创建了类的多个对象 每个对象都独立拥有一套类中的非静态属性。
*           当修改一个对象中的非金泰属性的时候，不会导致其他对象中同样的属性值的修改。
*
*           静态变量：我们创建类的多个对象，多个对象共享同一个静态变量。当通过某一个对象修改
*           这个变量的时候，会导致其他的变量也被修改了。
*
*    3。2 static 修饰属性的其他说明：
*       静态变量 随着类的加载而加载。可以通过 类.静态变量 的方式进行调用
*       静态变量的加载要早于对象的创建
*       由于类只会加载一次，则我们的静态变量在内存中只会存在一份。存在方法区的静态阈中
*
*            类变量     实例变量
*       类    yes        no
*       对象   yes        yes
*
*    3。3 静态属性举例：System.out   Math.Pi
*
*    4.使用static修饰方法：
*      4。1随着类的加载而加载,可以通过类。方法调用
*
*         4。2类方法     实例方法
 *   类     yes        no
 *   对象   yes        yes
*
*
*       4。3 静态方法中 只能调用静态的方法或属性
*       非静态方法中，既可以调用静态的 又能调用非静态的
*
*     5。static 注意点
*     5。1静态的方法内，不能用this 和super  因为这两个关键字只有对象实例生成的时候才能调用
*     可以用 类.属性 或者 类.方法
*
*     5。2 关于静态的属性和方法，大家都要从生命周期的角度去理解
*
*
*    6。在开发中，如何确定一个属性是否要声明为static
*    1。属性是可以被多个对象所共享的，不会随着对象的不同而不同
*
*
*    开发中，如何确定一个方法是否要声明为static
*    1。操作静态属性的方法，通常设置为静态的
*    2。工具类的方法 习惯上声明为 Static的 比如 Math Arrays
*
*
*
* */

public class StaticTest {
    public static void main(String[] args) {
        Chinese c1=new Chinese();
        c1.name="yaoMing";
        c1.age=40;

        Chinese c2=new Chinese();
        c2.name="maLong";
        c2.age=35;

        //
        c1.nation="CHN";
        System.out.println(c2.nation);

        Chinese.show();



    }
}


class Chinese{

    String name;
    int age;
    static String nation;

    public void eat()
    {
        System.out.println("chichicih");
    }

    public static void show()

        System.out.println("show");
    }



}
