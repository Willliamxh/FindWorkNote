题目：球队队伍排名

```
String[] team=new String[]{"a-b 3:0","a-c 2:1","b-a 1:1","c-a 0:1","b-c 4:3","c-b 2:2"};
```

先按照value进行排序 之后按照key进行排序

```java
String[] team=new String[]{"a-b 3:0","a-c 2:1","b-a 1:1","c-a 0:1","b-c 4:3","c-b 2:2"};

        HashMap<String,Integer> map=new HashMap();
        for(int i=0;i<team.length;i++)
        {
            String team1=team[i].substring(0,1);
            String team2=team[i].substring(2,3);
//            System.out.println(team1);;
//            System.out.println(team2);
            String team1Value=team[i].substring(4,5);
            String team2Value=team[i].substring(6,7);

//            System.out.println(team1Value);
//            System.out.println(team2Value);
            if(team1Value.compareTo(team2Value)>0)
            {
//                System.out.println("big");
                if(map.containsKey(team1))
                {
                    int valueTemp=map.get(team1)+3;
                    map.put(team1,valueTemp);
                }else
                {
                    map.put(team1,3);
                }

                if(!map.containsKey(team2))
                {
                    map.put(team2,0);
                }
            }
            else if(team1Value.compareTo(team2Value)<0)
            {
//                System.out.println("small");
                if(map.containsKey(team2))
                {
                    int valueTemp=map.get(team2)+3;
                    map.put(team2,valueTemp);
                }else
                {
                    map.put(team2,3);
                }

                if(!map.containsKey(team1))
                {
                    map.put(team1,0);
                }
            }
            else if(team1Value.compareTo(team2Value)==0)
            {
//                System.out.println("equal");
                if(map.containsKey(team1))
                {
                    int valueTemp=map.get(team1)+1;
                    map.put(team1,valueTemp);
                }else
                {
                    map.put(team1,1);
                }

                if(map.containsKey(team2))
                {
                    int valueTemp=map.get(team2)+1;
                    map.put(team2,valueTemp);
                }else
                {
                    map.put(team2,1);
                }
            }
//            System.out.println();
        }
        System.out.println(map);
        List<Map.Entry<String,Integer>> list = new ArrayList<>();
        list.addAll(map.entrySet());
        Collections.sort(list,new Comparator<Map.Entry<String,Integer>>(){
            public int compare(Map.Entry<String,Integer>e1,Map.Entry<String,Integer>e2){
                int re = e2.getValue().compareTo(e1.getValue());
                if(re!=0){return re;}
                else{return e1.getKey().compareTo(e2.getKey());}
            }
        });

        for(int i=0;i<list.size();i++){

            System.out.print(list.get(i).getKey()+":");
            System.out.println(list.get(i).getValue());
```
